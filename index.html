## 📋 配置步骤（必读） **Step 1**：在 Supabase Dashboard → SQL Editor 里粘贴并运行下面的 SQL（建表 + RLS + RPC 函数） **Step 2**：在 Dashboard → Authentication → Users → Add user，创建一个管理员账号（如 `admin@xianr.lab` / 自定义密码） **Step 3**：部署新 HTML 到 GitHub Pages 即可 --- ### 🗄️ SQL 初始化脚本 ```sql -- ============================================ -- 先研实验室问答平台 · Supabase Schema -- 一次性执行即可 -- ============================================ -- 1. 问题表 create table if not exists questions ( id uuid primary key default gen_random_uuid(), qid text unique not null, title text not null, content text not null, progress text[] not null default '{}', priority text not null default 'medium', status text not null default 'pending' check (status in ('pending','approved','rejected')), reject_reason text, session_id text not null, created_at timestamptz default now(), reviewed_at timestamptz ); -- 2. 回答表 create table if not exists answers ( id uuid primary key default gen_random_uuid(), question_id uuid not null references questions(id) on delete cascade, aid text unique not null, content text not null, nickname text default '匿名', adopted boolean default false, created_at timestamptz default now() ); create index if not exists idx_questions_status on questions(status); create index if not exists idx_questions_created on questions(created_at desc); create index if not exists idx_questions_session on questions(session_id); create index if not exists idx_answers_question on answers(question_id); -- ============================================ -- 3. RLS 策略 -- ============================================ alter table questions enable row level security; alter table answers enable row level security; -- 任何人可读已通过的问题 drop policy if exists "q_select_public" on questions; create policy "q_select_public" on questions for select using (status = 'approved'); -- 认证用户（管理员）可读所有问题 drop policy if exists "q_select_admin" on questions; create policy "q_select_admin" on questions for select to authenticated using (true); -- 任何人可插入问题（匿名强制 status=pending） drop policy if exists "q_insert" on questions; create policy "q_insert" on questions for insert with check ( (auth.role() = 'anon' and status = 'pending') or auth.role() = 'authenticated' ); -- 仅管理员可更新/删除 drop policy if exists "q_update" on questions; create policy "q_update" on questions for update to authenticated using (true); drop policy if exists "q_delete" on questions; create policy "q_delete" on questions for delete to authenticated using (true); -- 回答：读已通过问题的回答 + 管理员读全部 drop policy if exists "a_select" on answers; create policy "a_select" on answers for select using ( exists ( select 1 from questions q where q.id = answers.question_id and q.status = 'approved' ) or auth.role() = 'authenticated' ); -- 任何人可给已通过的问题写回答 drop policy if exists "a_insert" on answers; create policy "a_insert" on answers for insert with check ( exists ( select 1 from questions q where q.id = answers.question_id and q.status = 'approved' ) or auth.role() = 'authenticated' ); drop policy if exists "a_update" on answers; create policy "a_update" on answers for update to authenticated using (true); drop policy if exists "a_delete" on answers; create policy "a_delete" on answers for delete to authenticated using (true); -- ============================================ -- 4. RPC 函数（处理"我的问题"等场景） -- ============================================ -- 获取当前用户可见的问题（已通过 + 自己提交的） create or replace function get_my_questions(my_session text) returns setof questions language sql security definer as $$ select * from questions where status = 'approved' or session_id = my_session order by case priority when 'high' then 0 when 'medium' then 1 else 2 end, created_at desc; $$; -- 获取待审核数量（公开，用于导航栏 badge） create or replace function get_pending_count() returns integer language sql security definer as $$ select count(*)::int from questions where status = 'pending'; $$; -- 提问者采纳回答（验证 session 归属） create or replace function adopt_answer(p_answer_id uuid, my_session text) returns boolean language plpgsql security definer as $$ declare v_qid uuid; begin select a.question_id into v_qid from answers a join questions q on q.id = a.question_id where a.id = p_answer_id and q.session_id = my_session; if v_qid is null then return false; end if; update answers set adopted = false where question_id = v_qid; update answers set adopted = true where id = p_answer_id; return true; end; $$; -- 提问者撤回问题 create or replace function delete_own_question(p_question_id uuid, my_session text) returns boolean language plpgsql security definer as $$ begin if exists ( select 1 from questions where id = p_question_id and session_id = my_session and status != 'approved' ) then delete from questions where id = p_question_id; return true; end if; return false; end; $$; ``` --- ### 🌐 完整前端代码 ```html 先研实验室问答平台 · 一生一芯项目
